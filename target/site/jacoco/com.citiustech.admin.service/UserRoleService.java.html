package com.citiustech.admin.service;

import java.util.*;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import com.citiustech.admin.entity.UserRole;
import com.citiustech.admin.repository.UserRoleRepository;


@Service
public class UserRoleService {
	
	@Autowired
	private UserRoleRepository userRoleRepository;
	

	public boolean isRoleTypePresent(String roleType) {
		// TODO Auto-generated method stub
//		if(userRoleRepository.findByRoleType(roleType)!= null) { 
//			System.out.println("Role Type: " + roleType);
//			return true;}
//		else 
//			return false;
		return userRoleRepository.findByRoleType(roleType).isPresent();
		
	}

	
	public void save(UserRole user) {
		// TODO Auto-generated method stub
		System.out.println("inside save methos of user service Impl" + user.toString());
		userRoleRepository.save(user);
	}

	
	public Optional<List<UserRole>> getAllData() {
		// TODO Auto-generated method stub
		List<UserRole> userRole = userRoleRepository.findAll();
		return Optional.of(userRole);
	}

	
	public Optional<UserRole> getById(int id) {
		// TODO Auto-generated method stub
		return userRoleRepository.findByUserRoleId(id);
	}

	public Optional<UserRole> getByRoleType(String roleType) {
		// TODO Auto-generated method stub
		return userRoleRepository.findByRoleType(roleType);
	}


}
